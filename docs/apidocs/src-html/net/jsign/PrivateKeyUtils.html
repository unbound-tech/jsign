047/**
048 * Helper class for loading private keys (PVK or PEM, encrypted or not).
049 * 
050 * @author Emmanuel Bourg
051 * @since 2.0
052 */
053public class PrivateKeyUtils {
054
055    private PrivateKeyUtils() {
056    }
057
058    /**
059     * Load the private key from the specified file. Supported formats are PVK and PEM,
060     * encrypted or not. The type of the file is inferred from its extension (<code>.pvk</code>
061     * for PVK files, <code>.pem</code> for PEM files).
062     * 
063     * @param file     the file to load the key from
064     * @param password the password protecting the key
065     * @return the private key loaded
066     * @throws KeyException if the key cannot be loaded
067     */
068    public static PrivateKey load(File file, String password) throws KeyException {
069        try {
070            if (file.getName().endsWith(".pvk")) {
071                return PVK.parse(file, password);
072            } else if (file.getName().endsWith(".pem")) {
073                return readPrivateKeyPEM(file, password);
074            }
075        } catch (Exception e) {
076            throw new KeyException("Failed to load the private key from " + file, e);
077        }
078        
079        throw new IllegalArgumentException("Unsupported private key format (PEM or PVK file expected");
080    }
081
082    private static PrivateKey readPrivateKeyPEM(File file, String password) throws IOException, GeneralSecurityException, OperatorCreationException, PKCSException {
083        try (FileReader reader = new FileReader(file)) {
084            PEMParser parser = new PEMParser(reader);
085            Object object = parser.readObject();
086            
087            if (object == null) {
088                throw new IllegalArgumentException("No key found in " + file);
089            }
090            
091            BouncyCastleProvider provider = new BouncyCastleProvider();
092            JcaPEMKeyConverter converter = new JcaPEMKeyConverter().setProvider(provider);
093
094            if (object instanceof PEMEncryptedKeyPair) {
095                // PKCS1 encrypted key
096                PEMDecryptorProvider decryptionProvider = new JcePEMDecryptorProviderBuilder().setProvider(provider).build(password.toCharArray());
097                PEMKeyPair keypair = ((PEMEncryptedKeyPair) object).decryptKeyPair(decryptionProvider);
098                return converter.getPrivateKey(keypair.getPrivateKeyInfo());
099
100            } else if (object instanceof PKCS8EncryptedPrivateKeyInfo) {
101                // PKCS8 encrypted key
102                InputDecryptorProvider decryptionProvider = new JceOpenSSLPKCS8DecryptorProviderBuilder().setProvider(provider).build(password.toCharArray());
103                PrivateKeyInfo info = ((PKCS8EncryptedPrivateKeyInfo) object).decryptPrivateKeyInfo(decryptionProvider);
104                return converter.getPrivateKey(info);
105                
106            } else if (object instanceof PEMKeyPair) {
107                // PKCS1 unencrypted key
108                return converter.getKeyPair((PEMKeyPair) object).getPrivate();
109                
110            } else if (object instanceof PrivateKeyInfo) {
111                // PKCS8 unencrypted key
112                return converter.getPrivateKey((PrivateKeyInfo) object);
113                
114            } else {
115                throw new UnsupportedOperationException("Unsupported PEM object: " + object.getClass().getSimpleName());
116            }
117        }
118    }
